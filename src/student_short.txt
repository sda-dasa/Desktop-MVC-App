class StudentShort
	def initialize (id, last_name_initials, contact, git)
		@id = id
		@last_name_initials  = last_name_initials
		@contact = contact
		@git = git
	end

	attr_reader :id, :last_name_initials, :contact, :git

	def self.from_student (student)
		if not student.has_id?
			puts "у данного студента отсутсвует id - не достаточно данных!"
		else
			new (student.id, student.last_name_initials, student.contact, student.git)
		end
	end

	def to_s
		return "id: #{@id} last_name_initials: #{@last_name_initials} 
contact: #{@contact} git: #{@git}"
	end

	def has_git?
		if git.nil? or git.empty? 
			return false
		end
		return true
	end

	def has_contact?
		if contact.nil? or contact.empty? 
			return false
		end
		return true	
	end



end
